✅ 1. Enable bucketing
SET hive.enforce.bucketing = true;
✅ 2. Create bucket table
CREATE TABLE st_bucket(
  roll_no INT,
  name STRING,
  subject STRING,
  marks INT,
  section STRING
)
CLUSTERED BY (roll_no) INTO 3 BUCKETS
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ',';
✅ 3. Insert data into bucket table

⚠️ Important: Use INSERT SELECT, not direct load

INSERT INTO TABLE st_bucket
SELECT * FROM student_marks;
✅ 4. Check data
SELECT * FROM st_bucket;
✅ 5. Verify bucketing (important for viva)
