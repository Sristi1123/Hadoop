// Create an empty file
    hdfs dfs -touchz /PROJECT/f1

// Append from keyboard (stdin) to an HDFS file
    hdfs dfs -appendToFile - /PROJECT/f1

// Append a local file's content to an HDFS file
    hdfs dfs -appendToFile "D:\hadoopsetup\data.txt" /PROJECT/f1

// Read / Print an HDFS file to the terminal
    hdfs dfs -cat /PROJECT/f1

// Check file size / disk usage
    # Show size of a specific file
    hdfs dfs -du -h /PROJECT/f1

    # Show size of all files inside a directory
    hdfs dfs -du -h /PROJECT
